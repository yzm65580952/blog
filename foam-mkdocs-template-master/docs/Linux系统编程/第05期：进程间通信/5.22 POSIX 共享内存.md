```c
#include <stdio.h>
#include <unistd.h>
#include <semaphore.h>
#include <sys/mman.h>
#include <sys/stat.h>
#include <string.h>
#include <fcntl.h>
#include <errno.h>

#define SHM_NAME "/shm"	//定义共享内存的文件名

int main (void)
{
	int shm_fd;	//1.定义文件描述符
	shm_fd = shm_open (SHM_NAME, O_RDWR | O_CREAT, 0666);
	//2.打开共享内存
	if (shm_fd < 0)
	{
		perror ("shm_open");
		return -1;
	}

	ftruncate (shm_fd, 8192);//3.调整文件大小
    struct stat filestat;
	fstat (shm_fd, &filestat);//4.获取共享内存的信息
    printf ("st_size: %ld\n", filestat.st_size);

	char *shm_ptr;
	shm_ptr = (char*)mmap (NULL, filestat.st_size, \
              PROT_READ|PROT_WRITE, MAP_SHARED, shm_fd, 0);
			  //5.映射共享内存 NULL用户进程会自己找合适的区域
			  //shm_fd指向要映射的物理内存
	close (shm_fd);//6.关闭shm_fd

	char buf[] = "hello world";
	memmove (shm_ptr, buf, sizeof (buf));//7.向共享内存写字符串
	printf ("pid %d: %s\n", getpid(), shm_ptr);
    munmap (shm_ptr, filestat.st_size);//8.取消映射
	return 0;
}
```

## 通过信号量同步共享内存
wrire_shm_sem.c
```c
/************************************************
*      Filename: wrire_shm_sem.c
*        Author: litao.wang

*   Description: 
*        Create: 2019-07-03 22:07:06
* Last Modified: 2019-07-22 17:52:05
*************************************************/
#include <stdio.h>
#include <unistd.h>
#include <semaphore.h>
#include <sys/mman.h>
#include <sys/stat.h>
#include <string.h>
#include <fcntl.h>
#include <errno.h>

#define SHM_NAME "/memmap"
#define SEM_NAME "/memmap_sem"

int main (void)
{
	int shm_fd;
	sem_t *sem;
	shm_fd = shm_open (SHM_NAME, O_RDWR | O_CREAT, 0666);
	//1.打开一片共享内存
	sem = sem_open (SEM_NAME, O_CREAT, 0666, 0);
	//2.打开信号量
	if (shm_fd < 0 || sem == SEM_FAILED)
	{
		perror ("open");
		return -1;
	}
    ftruncate (shm_fd, 8192);//3.设置共享内存的大小
	struct stat filestat;
	fstat (shm_fd, &filestat);
    printf ("st_size: %ld\n", filestat.st_size);
	
	char *shm_ptr;
	shm_ptr = (char*)mmap (NULL, filestat.st_size,  \
                PROT_READ|PROT_WRITE, MAP_SHARED, shm_fd, 0);
				//4.映射共享内存
	close (shm_fd);//5.关闭shm_fd

	char msg[] = "hello world";
	memmove (shm_ptr, msg, sizeof (msg));	//6.写共享内存
	printf ("pid %d: %s\n", getpid(), shm_ptr);
    
	sem_post (sem);	//7.信号量+1，提醒读进程读
	sem_close (sem);	//8.关闭信号量
    sleep (5);
    munmap (shm_ptr, 8192);//9.取消内存映射

	return 0;
}
```

```c
#include <stdio.h>
#include <unistd.h>
#include <stdlib.h>
#include <semaphore.h>
#include <sys/mman.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <errno.h>

#define SHM_NAME "/memmap"
#define SEM_NAME "/memmap_sem"

int main (void)
{
	int shm_fd;
	sem_t *sem;
	shm_fd = shm_open (SHM_NAME, O_RDWR|O_CREAT, 0666);
	//1.打开一片共享内存
	sem = sem_open (SEM_NAME, 0);
	//2.打开信号量
	if (shm_fd < 0 || sem == SEM_FAILED)
	{
		perror ("open");
		return -1;
	}
    ftruncate (shm_fd, 8192);//3.设置共享内存的大小
	struct stat filestat;
	fstat (shm_fd, &filestat);
    printf ("st_size: %ld\n", filestat.st_size);
	char *shm_ptr;
	shm_ptr = (char*)mmap (NULL, filestat.st_size, PROT_READ|PROT_WRITE, \
							MAP_SHARED, shm_fd, 0);
							//4.映射共享内存
    if (shm_ptr ==(void*)-1)
    {
        perror ("mmap");
        exit (-1);
    }
	close (shm_fd);//5.关闭shm_fd
	sem_wait (sem);//6.信号量-1
	
	printf ("pid %d: %s\n", getpid(), shm_ptr);//7.读共享内存
	sem_close (sem);
    munmap (shm_ptr, 8192);//8.取消内存映射
    shm_unlink (SHM_NAME);//7.删除共享内存
    sem_unlink (SEM_NAME);//9.删除信号量
	return 0;
}
```
