#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <unistd.h>
#include <fcntl.h>

struct ctl_cmd_info{		//分支机构体，依据该变量来执行
    int if_result;
    int if_block;
    int else_block;
};

int is_control_cmdline (char buffer[])// 判断是if-else流程控制
{
    return (strncmp (buffer, "if", 2) == 0   || \
            strncmp (buffer, "else", 4) == 0 || \
            strncmp (buffer, "fi", 2) == 0);
}

int is_exec_cmdline (struct ctl_cmd_info cmd_info)
{
    if ((cmd_info.if_result && cmd_info.if_block)          ||  \
        ((cmd_info.if_result == 0) && cmd_info.else_block) || \
        ((cmd_info.if_block == 0) && (cmd_info.else_block == 0)))
        return 1;
    else
        return 0;
}

int is_if_cmdline_true (char buffer[])
{
    char *arg_array[10] = {0};
    char *cp = buffer;
    int len = 0;
    char c;
    int args_num = 0;
    char *start;
    while (*cp != '[')
        cp++;
    cp++;
    while (strncmp (cp, "]", 1) != 0)//解析[]里面的内容
    {
        while (*cp == ' ')
            cp++;
        if (strncmp (cp, "]", 1) == 0)
            break;
        len = 1;
        start = cp;
        c = *++cp;
        while (c != ' ' && c != '\t')
        {
            len++;
            c = *++cp;
        }
        arg_array[args_num] = (char *)malloc (len);
        strncpy (arg_array[args_num++], start, len);
		//解析出来的token的地址存放在arg_array[]里面
    }
    if (strncmp (arg_array[0], "-f", 2) == 0)//参数-f 
    {
        int result = 0;
        if ((access (arg_array[1], F_OK)) != -1)
		//检测文件是否存在
            result = 1;
        free (arg_array[0]);
        free (arg_array[1]);
        return result;
    }
    if (strncmp (arg_array[0], "-d", 2) == 0)//参数-d
    {
        int result = 0;
        if ((access (arg_array[1], F_OK)) != -1)
		//检测目录是否存在
            result = 1;
        free (arg_array[0]);
        free (arg_array[1]);
        return result;
    }

    return 0;
}
 
void process_control_cmdline (char buf[], struct ctl_cmd_info *cmd_info)
{
    if (strncmp (buf, "if", 2) == 0)
    {
        cmd_info->if_block = 1;
        cmd_info->else_block = 0;
        if (is_if_cmdline_true (buf))//判断if 的内容是否为真
            cmd_info->if_result = 1;
    }
    if (strncmp (buf, "else", 4) == 0)
    {
        cmd_info->if_block = 0;
        cmd_info->else_block = 1;
    }
    if (strncmp (buf, "fi", 2) == 0)
    {
        cmd_info->if_block = 0;
        cmd_info->else_block = 0;
        cmd_info->if_result = 0;
    }

}

    1: background exec mode

int main (int argc, char *argv[])

void print_cmd_prompt (void)
{
    struct passwd *pwd;
    pwd = getpwuid (getuid());
    char path[256];
    getcwd (path, sizeof(path));
    printf ("\033[33;1m%s@wsh:\033[34;1m%s\033[0m >>>", pwd->pw_name, path);
}

int read_cmdline (char buffer[])
{
    char c;
    int pos = 0;
    while (1)
    {
        c = getchar ();
        if (c != '\n' && c != EOF)
            buffer[pos++] = c;
        else
        {
            buffer[pos++] = '\0';
            break;
        }
    }
    return pos;
}

int parse_cmdline (char buffer[], char *arglist[])
{
    int  args_num = 0;
    char *cp = buffer;
    int  len;
    char *start;
    char c;
    while (*cp != '\0' && *cp != '\n')
    {
        if (*cp == '&')
        {
            background_exec = 1;
            *cp = '\0';
            break;
        }
        cp++;
    }

    cp = buffer;
    while (*cp != '\0')
    {
        while (*cp == ' ' || *cp == '\t')
            cp++;
        if (*cp == 0)
            break;
        start = cp;
        len = 1;
        c = *++cp;
        while (c != '\0' && c != ' ' && c != '\t')
        {
            len++;
            c = *++cp;
        }
        arglist[args_num] = (char *) malloc (len + 1);
        strncpy (arglist[args_num], start, len);
        arglist[args_num][len] = '\0';
        args_num++;
    }
    return args_num;
}
int is_builtin_cmd (char *arglist[]);
int exec_builtin_cmd (char *arglist[]);

int exec_cmd (char *arglist[], int fd_in, int fd_out)
{
    int ret_from_fork;
    int child_ret_status = 0;
    ret_from_fork = fork ();
    if (ret_from_fork == 0)
    {
        signal (SIGINT, SIG_DFL);
        signal (SIGQUIT, SIG_DFL);

        dup2 (fd_in, 0);
        dup2 (fd_out, 1);
        if (fd_in != 0)
            close (fd_in);
        if (fd_out != 1)
            close (fd_out);

        execvp (arglist[0], arglist);
        perror ("execvp");
        exit (EXIT_FAILURE);
    }
    else if (ret_from_fork > 0)
    {
        signal (SIGINT, SIG_IGN);
        signal (SIGQUIT, SIG_IGN);
        if (background_exec == 1)
        {
            background_exec = 0;
            signal (SIGCHLD, SIG_IGN);
        }
        else
        {
            signal (SIGCHLD, SIG_DFL);
            if(wait (&child_ret_status) == -1)
                perror ("wait");
        }

        if (fd_in != 0)
            close (fd_in);
        if (fd_out != 1)
            close (fd_out);
    }
    else
    {
        perror ("fork");
        exit (EXIT_FAILURE);
    }
    
    return child_ret_status;
}

int is_pipe_command (char buffer[])
{
    char *cp = buffer;
    while (*cp != '\n' && *cp != '\0')
    {
        if (strncmp (cp, "|", 1) == 0)
            return 1;
        cp++;
    }
    return 0;
}

int exec_cmdline (char *arglist[], char buffer[])
{
    char *temp = buffer;
    char *args;
    int  args_num = 0;
    int fd_tmp = 0;

    pipe_mode = is_pipe_command (buffer);

    while ((args = strtok(temp, "|")) != NULL)
    {
        temp = NULL;
        args_num = parse_cmdline (args, arglist);
        if (arglist[0] == 0)
            return 0;
        if (is_builtin_cmd(arglist))
        {
            exec_builtin_cmd (arglist);
            return 0;
        }
        if (pipe_mode == 1)
        {
            int fd[2];
            if (pipe(fd) == -1)
            {
                perror ("pipe");
                exit (EXIT_FAILURE);
            }
            exec_cmd (arglist, fd_tmp, fd[1]);
            fd_tmp = fd[0];
            pipe_mode = 0;
        }
        else
        {
            exec_cmd (arglist, fd_tmp, 1);
        }
        free_cmdline (arglist, args_num);
        memset (arglist, 0, 80);
    }
    fd_tmp = 0;
    pipe_mode = 0;
    return 0;
}

int exec_script (char pathname[], char buffer[], char *arglist[])
{
    FILE *fp;
    fp = fopen (pathname, "r");
    if (fp == NULL)
    {
        perror ("fopen");
        exit (EXIT_FAILURE);
    }
    char c;
    int pos = 0;
    struct ctl_cmd_info cmd_info = {0,0,0};
    while (1)
    {
        c = getc (fp);
        if (c == '\n' || c == EOF)
        {
            buffer[pos++] = '\0';
            if (strlen (buffer) == 1 && c == '\n')
                continue;
            if (is_control_cmdline (buffer))//如果是控制字符串
            {
                process_control_cmdline (buffer, &cmd_info);
                memset (buffer, '\0', 512);
				//清理buffer，准备下一次
                pos = 0;
                continue;
            }
            if (is_exec_cmdline (cmd_info))
                exec_cmdline (arglist, buffer);
            memset (buffer, '\0', 512);
            memset (arglist, 0, 80);
            pos = 0;
            if (c == EOF) //check end of file
                break;
        }
        else
            buffer[pos++] = c;
    }
    fclose (fp);
    return 0;
}

int free_cmdline (char *arglist[], int args_num)
{
    int i;
    for (i = 0; i < args_num; i++)
        free (arglist[i]);
    return 0;
}
